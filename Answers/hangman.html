from random_word import RandomWords

#gets a random word, returns in the form of _ _ _ _ ...
def generate_word():
    r = RandomWords()
    word = r.get_random_word()
    ret = ""
    for i in word:
        ret += " " + i
    return ret

def generate_blanks(word):
    ret = ""
    for i in word:
        if i == 0:
            ret += "_"
        elif i != " ":
            ret += " _"
    return ret

guessed_letters = []
word_to_guess = generate_word()
print(word_to_guess)
display_word = generate_blanks(word_to_guess)
while display_word != word_to_guess:
    printable = ""
    for word in guessed_letters:
        printable += word
    
    print("GUESSED LETTERS: " + printable)
    print(display_word)
    if(len(printable) >= 10):

    guess = input("Please enter a letter.")
    guessed_letters.append(guess)
    
    new_display = ""

        if word_to_guess[i] == guess[0] and i == 0:
            new_display += guess[0]
        elif word_to_guess[i] == guess[0]:
            new_display += guess[0]
        else:
            new_display += display_word[i]
    display_word = new_display
