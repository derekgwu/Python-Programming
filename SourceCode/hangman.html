from random_word import RandomWords

#gets a random word, returns in the form of _ _ _ _ ...
def generate_word():
    r = RandomWords()
    word = r.get_random_word()
    ret = ""
    for i in word:
        ret += " " + i
    return ret

def generate_blanks(word):
    ret = ""
    for i in word:
        if i == 0:
            ret += "_"
        elif i != " ":
            ret += " _"
    return ret
